function selectTag(tagElement) { const tag = tagElement.getAttribute('data-tag'); tagElement.classList.toggle('selected'); // 선택된 태그를 폼의 숨겨진 입력 필드에 추가 const selectedTagsInput = document.getElementById('selectedTags'); let selectedTags = selectedTagsInput.value.split(',').filter(Boolean); if (tagElement.classList.contains('selected')) { if (!selectedTags.includes(tag)) { selectedTags.push(tag); } } else { selectedTags = selectedTags.filter(t => t !== tag); } selectedTagsInput.value = selectedTags.join(','); }
